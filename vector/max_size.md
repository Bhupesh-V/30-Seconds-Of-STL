// Creating the vector
std::vector<int>vector;
	
for (int i=0; i<100; ++i) 
vector.push_back(i);

// Outputting the vectors size, capacity, and max size
std::cout << "size: " << vector.size() << "\n";
std::cout << "capacity: " << vector.capacity() << "\n";
std::cout << "max_size: " << vector.max_size() << "\n";
